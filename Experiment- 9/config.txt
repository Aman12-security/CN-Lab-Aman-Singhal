# Step 1: List block devices
lsblk

# Step 2: Display detailed disk information
sudo fdisk -l

# Step 3: Create a mount point
sudo mkdir /mnt/usb

# Step 4: Mount the storage device (example: /dev/sdb1)
sudo mount /dev/sdb1 /mnt/usb

# Step 5: Verify mounted devices
df -h
mount

# Step 6: Access the mounted device
cd /mnt/usb
ls

# Step 7: Exit the mounted directory
cd ~

# Step 8: Unmount the device
sudo umount /mnt/usb

# Step 9: Verify unmounting
df -h
